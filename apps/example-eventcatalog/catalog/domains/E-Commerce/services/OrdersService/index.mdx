## Overview

The Orders Service is the entry point of the checkout. It validates the incoming
order, persists it, and publishes [[event|OrderPlaced]] on the
[[channel|orders.events]] Kafka topic.

It listens for [[event|PaymentCaptured]] and [[event|InventoryReserved]] to
advance the order to `confirmed`.

### How it's instrumented

The order acceptance handler is wrapped in `trace()`. The Kafka publish uses
`traceProducer` so the consumer-side spans link automatically.

```ts
import { trace, track } from 'autotel';
import { traceProducer } from 'autotel/messaging';

const publishOrderPlaced = traceProducer({
  system: 'kafka',
  destination: 'orders.events',
})(() => async (msg) => kafka.publish('orders.events', msg));

export const placeOrder = trace((ctx) => async (order: PlaceOrderInput) => {
  ctx.setAttribute('order.customer_id', order.customerId);
  ctx.setAttribute('order.value_cents', order.totalCents);

  await db.orders.insert(order);
  await publishOrderPlaced({ type: 'OrderPlaced', ...order });

  track('order.placed', {
    orderId: order.id,
    customerId: order.customerId,
    totalCents: order.totalCents,
    _autotel: { channel: 'orders.events', producer: 'OrdersService' },
  });

  return order;
});
```
