## Overview

The Inventory Service consumes [[event|PaymentCaptured]] and reserves stock in
the warehouse management system. Once the reservation is acknowledged it emits
[[event|InventoryReserved]] back to the Orders Service.

<NodeGraph />

### How it's instrumented

The reservation crosses a service boundary (WMS), so it's wrapped in `trace()`
with a child `span()` for the WMS call. Both inherit the consumer span from
`traceConsumer` so the producer→consumer→WMS chain is one trace.

```ts
import { trace, span, track } from 'autotel';
import { traceConsumer } from 'autotel/messaging';

export const reserveStock = trace((ctx) => async (orderId: string, items: Item[]) => {
  ctx.setAttribute('inventory.item_count', items.length);

  const reservation = await span('wms.reserve', async (s) => {
    s.setAttribute('wms.system', 'manhattan');
    return wms.reserve(items);
  });

  track('inventory.reserved', {
    orderId,
    reservationId: reservation.id,
    _autotel: { channel: 'inventory.events', producer: 'InventoryService' },
  });
  return reservation;
});

export const handlePaymentCaptured = traceConsumer({
  system: 'kafka',
  destination: 'payments.events',
})(() => async (msg) => reserveStock(msg.orderId, msg.items));
```
